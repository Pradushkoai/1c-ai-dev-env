"""
tools/ — определения MCP tools по категориям.

P1.3: вынесено из src/mcp_server.py для декомпозиции (SRP).

Каждый модуль экспортирует:
- DESCRIPTIONS: list[dict] — статическое описание tools для _get_tools_description()

Категории:
- search: поиск по методам платформы и коду конфигурации
- analyzers: анализ BSL кода (11 анализаторов)
- metadata: метаданные конфигурации (объекты, формы, СКД)
- generate: генерация кода (обработки, отчёты, EPF)
- dsl_cfe: DSL компиляторы и работа с CFE
- quality: проверка качества (формы, СКД, diff)
- context: сбор контекста (solve, knowledge, data_status)
- misc: прочие (inspect, openspec, depgraph, skd_trace)
"""

def get_all_descriptions() -> list[dict]:
    """Получить все описания tools из всех категорий.

    Returns:
        Объединённый список описаний 45 tools.
    """
    all_descriptions: list[dict] = []
    all_descriptions.extend(SEARCH_DESCRIPTIONS)
    all_descriptions.extend(ANALYZERS_DESCRIPTIONS)
    all_descriptions.extend(METADATA_DESCRIPTIONS)
    all_descriptions.extend(GENERATE_DESCRIPTIONS)
    all_descriptions.extend(DSL_CFE_DESCRIPTIONS)
    all_descriptions.extend(QUALITY_DESCRIPTIONS)
    all_descriptions.extend(CONTEXT_DESCRIPTIONS)
    all_descriptions.extend(MISC_DESCRIPTIONS)
    return all_descriptions
